部署方案

🐳 Docker Compose (开发环境)

version: '3.8'

services:
  backend:
    build: ./backend
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://user:pass@db:5432/rdautomation
      - REDIS_URL=redis://redis:6379
  
  frontend:
    build: ./frontend
    ports:
      - "3000:80"
  
  db:
    image: postgres:15-alpine
    volumes:
      - postgres_data:/var/lib/postgresql/data
  
  redis:
    image: redis:7-alpine

☸️ Kubernetes (生产环境)

apiVersion: apps/v1
kind: Deployment
metadata:
  name: rd-automation-backend
spec:
  replicas: 3
  selector:
    matchLabels:
      app: rd-automation
  template:
    spec:
      containers:
      - name: backend
        image: registry.example.com/rd-automation:latest
        resources:
          requests:
            memory: "512Mi"
            cpu: "250m"
          limits:
            memory: "1Gi"
            cpu: "500m"
