A.1 Dockerfile 完整示例

Dockerfile · 多层优化构建
# 阶段 1: 基础镜像（Python 3.12 + 系统依赖）
FROM python:3.12-slim-bookworm AS base

# 设置环境变量
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1
ENV PIP_NO_CACHE_DIR=1
ENV PIP_DISABLE_PIP_VERSION_CHECK=1

# 安装系统依赖
RUN apt-get update && apt-get install -y --no-install-recommends \
    curl \
    git \
    ffmpeg \
    libmagic1 \
    && rm -rf /var/lib/apt/lists/*

# 阶段 2: 依赖安装（利用 Docker 缓存层）
FROM base AS dependencies

WORKDIR /app

# 复制依赖文件并安装
COPY requirements.txt .
RUN pip install --user --no-cache-dir -r requirements.txt

# 阶段 3: 构建应用
FROM base AS builder

WORKDIR /app

# 复制已安装的依赖
COPY --from=dependencies /root/.local /root/.local
COPY --from=dependencies /usr/local/lib/python3.12/site-packages /usr/local/lib/python3.12/site-packages

# 复制应用代码
COPY . .

# 阶段 4: 最终运行镜像（最小化）
FROM base AS runtime

ARG APP_USER=appuser
ARG APP_GROUP=appgroup
ARG APP_UID=1000
ARG APP_GID=1000

# 创建非 root 用户
RUN groupadd -g ${APP_GID} ${APP_GROUP} && \
    useradd -u ${APP_UID} -g ${APP_GROUP} -m -s /bin/bash ${APP_USER}

WORKDIR /app

# 复制构建产物
COPY --from=builder --chown=${APP_USER}:${APP_GROUP} /app .
COPY --from=builder --chown=${APP_USER}:${APP_GROUP} /root/.local /home/${APP_USER}/.local

# 设置环境变量
ENV PATH=/home/${APP_USER}/.local/bin:$PATH
ENV HOME=/home/${APP_USER}

# 切换到非 root 用户
USER ${APP_USER}

# 暴露端口
EXPOSE 8000

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:8000/health || exit 1

# 启动命令
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "4"]
