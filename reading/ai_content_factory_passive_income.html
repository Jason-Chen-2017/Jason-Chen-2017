A.2 集群软件部署流程

deploy_cluster.sh · 批量部署脚本
#!/bin/bash
# AI 内容工厂 Mac 集群批量部署脚本

# 定义 Mac 节点 IP 列表
NODES=("192.168.1.101" "192.168.1.102" "192.168.1.103" "192.168.1.104" "192.168.1.105"
       "192.168.1.106" "192.168.1.107" "192.168.1.108" "192.168.1.109" "192.168.1.110")

# SSH 密钥认证（提前生成并分发公钥）
SSH_KEY="~/.ssh/id_rsa"
SSH_USER="admin"

# 安装 Homebrew（如果未安装）
install_homebrew() {
    ssh -i $SSH_KEY $SSH_USER@$1 "/bin/bash -c \"\$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)\""
}

# 安装依赖软件
install_dependencies() {
    ssh -i $SSH_KEY $SSH_USER@$1 "brew install python3 node ffmpeg docker ansible"
}

# 克隆项目代码
clone_repo() {
    ssh -i $SSH_KEY $SSH_USER@$1 "git clone https://github.com/your-org/ai-content-factory.git ~/content-factory"
}

# 安装 Python 依赖
install_python_deps() {
    ssh -i $SSH_KEY $SSH_USER@$1 "cd ~/content-factory && pip3 install -r requirements.txt"
}

# 配置环境变量
setup_env() {
    ssh -i $SSH_KEY $SSH_USER@$1 "cp ~/content-factory/.env.example ~/content-factory/.env"
}

# 启动 Docker 容器
start_containers() {
    ssh -i $SSH_KEY $SSH_USER@$1 "cd ~/content-factory && docker-compose up -d"
}

# 批量执行（并行）
echo "🚀 开始部署 AI 内容工厂集群..."

for node in "${NODES[@]}"; do
    echo "部署节点：$node"
    install_homebrew $node &
    install_dependencies $node &
    clone_repo $node &
done

wait # 等待所有后台任务完成

for node in "${NODES[@]}"; do
    install_python_deps $node
    setup_env $node
    start_containers $node
    echo "✅ 节点 $node 部署完成"
done

echo "🎉 集群部署完成！共 ${#NODES[@]} 个节点"

部署步骤说明

  1. 准备工作: 所有 Mac 安装 macOS Monterey+，开启 SSH 远程登录
  2. 生成 SSH 密钥: ssh-keygen -t rsa，将公钥分发到所有节点
  3. 运行部署脚本: bash deploy_cluster.sh，约 15 分钟完成全部节点
  4. 验证集群: 访问主控面板（http://192.168.1.100:8080），查看所有节点状态
  5. 开始生产: 在控制面板创建任务，集群自动分配负载
